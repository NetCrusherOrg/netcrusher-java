    NioReactor reactor = new NioReactor();
    TcpCrusher crusher = TcpCrusherBuilder.builder()
        .withReactor(reactor)
        .withBindAddress("localhost", 10080)
        .withConnectAddress("google.com", 80)
        .buildAndOpen();

    // emulate reconnect
    crusher.crush();

    // freeze
    crusher.freeze();

    // unfreeze
    crusher.unfreeze();

    // close
    crusher.close();
    reactor.close();
